1function [RpeakPos Rpeak ECG_d ECG_i ECG_s] = fixed_rivas(ECG)
 2%function [RpeakPos Rpeak ] = fixed_rivas(ECG)
 3%ECG has to be long 21600 = 1 minute of recordings
 4%initaliazing
 5len=21600;
 6ECG_d=zeros(1,len);
 7ECG_i=zeros(1,len);
 8ECG_s=zeros(1,len);
 9
10%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
11%%DIFFERENTIATOR
12%difference equation: y(n)= x[n] -x[n-Nd]
13Nd = 7; %round(3*fs/128) -1;
14bd = [1 zeros(1,Nd-1) -1];
15zd = zeros(length(bd),1);
16[ECG_d1 zd1] = fir(bd,ECG(1:Nd),zd); %get initial conditions accounting for delay
17ECG_d = fir(bd,ECG(Nd:len),zd1);
18%
19%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
20%%MOVING WINDOW INTEGRATOR
21%%% Moving average y[n] =  1/(N-1) \Sigma_{k=0}^{N-1} x[n-k]
22N = Nd + 1; %window of the moving average
23%b = (1/(N-1))*ones(1,N);; %this is the one on the paper but I think it's wrong
24bi = (1/N)*ones(1,N);;
25zi = zeros(length(bi),1);
26ECG_i = fir(bi,ECG_d,zi);
27
28%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
29%%%SQUARING
30ECG_s = ECG_i.*ECG_i;
31%
32%%end of preprocessing stage
33%
34%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
35%PEAK Detection stage
36%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%%
37x=ECG;
38y = ECG_s; %output of the preprocessing stage
39%RRmin = 200e-3*fs;
40RRmin = 72;
41QRSint=21.6; %60e-3*fs;
42%Pth= 0.7*fs/128 + 4.7;
43K=0.98164; %6214845220;
44
45i = 1; %signal index
46r = 1; %R peaks and positions index
47th=0; %threshold
48count = 0;
49Rpeak = zeros(1,len);
50RpeakPos = Rpeak;
51%Rpeak=[];
52%RpeakPos=[];
53while (i < len-Nd+1)
54        %State 1
55        count = 0;
56        peak = y(i);
57        peakPos = i;
58        while (count <= RRmin + QRSint)
59                if(y(i) > peak)
60                        peak = y(i);
61                        peakPos = i;
62                end
63                count = count + 1;
64                i = i + 1;
65                if(i >= len-Nd+1) %to avoid going out of bounds
66                        break;
67                end
68        end
69        Rpeak(r) = peak;
70        RpeakPos(r) = peakPos;
71        count = i - RpeakPos(r) - 1; %to compensate
72        r = r + 1;
73
74        %State 2
75        while (count <= RRmin)
76                count = count + 1;
77        end
78        th = mean(Rpeak);
79
80        %State 3
81        while(y(i) <= th)
82                i = i + 1;
83                if(i >= len-Nd+1) %to avoid going out of bounds
84                        break;
85                end
86                th = th * K;
87        end
88end
89end
